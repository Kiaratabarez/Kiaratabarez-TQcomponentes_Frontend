-- Base de datos para TQComponents
CREATE DATABASE IF NOT EXISTS tqcomponents_db
CHARACTER SET utf8mb4
COLLATE utf8mb4_unicode_ci;

USE tqcomponents_db;


CREATE TABLE categorias (
    id_categoria INT AUTO_INCREMENT PRIMARY KEY,
    nombre_categoria VARCHAR(50) NOT NULL UNIQUE,
    descripcion TEXT,
    activo BOOLEAN DEFAULT TRUE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_modificacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE productos (
    id_producto INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    descripcion TEXT,
    precio DECIMAL(10, 2) NOT NULL,
    imagen VARCHAR(255),
    id_categoria INT NOT NULL,
    stock INT DEFAULT 0,
    destacado BOOLEAN DEFAULT FALSE,
    activo BOOLEAN DEFAULT TRUE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_modificacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (id_categoria) REFERENCES categorias(id_categoria) ON DELETE RESTRICT,
    INDEX idx_categoria (id_categoria),
    INDEX idx_destacado (destacado),
    INDEX idx_precio (precio)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;


CREATE TABLE usuarios (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    nombre_completo VARCHAR(150),
    telefono VARCHAR(20),
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ultimo_login TIMESTAMP NULL,
    activo BOOLEAN DEFAULT TRUE,
    
    INDEX idx_username (username),
    INDEX idx_email (email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;


CREATE TABLE direcciones_envio (
    id_direccion INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT NOT NULL,
    direccion VARCHAR(255) NOT NULL,
    ciudad VARCHAR(100) NOT NULL,
    codigo_postal VARCHAR(10) NOT NULL,
    pais VARCHAR(50) NOT NULL,
    predeterminada BOOLEAN DEFAULT FALSE,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    INDEX idx_usuario (id_usuario)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE pedidos (
    id_pedido INT AUTO_INCREMENT PRIMARY KEY,
    numero_pedido VARCHAR(20) NOT NULL UNIQUE,
    id_usuario INT NOT NULL,
    
    -- Información cliente
    nombre_cliente VARCHAR(100) NOT NULL,
    apellido_cliente VARCHAR(100) NOT NULL,
    email_cliente VARCHAR(100) NOT NULL,
    telefono_cliente VARCHAR(20) NOT NULL,
    
    -- Dirección del envío
    direccion_envio VARCHAR(255) NOT NULL,
    ciudad VARCHAR(100) NOT NULL,
    codigo_postal VARCHAR(10) NOT NULL,
    pais VARCHAR(50) NOT NULL,
    
    -- Información del pedido
    subtotal DECIMAL(10, 2) NOT NULL,
    costo_envio DECIMAL(10, 2) DEFAULT 0.00,
    total DECIMAL(10, 2) NOT NULL,
    
    -- Información de pago
    metodo_pago VARCHAR(50) NOT NULL,
    estado_pago ENUM('pendiente', 'pagado', 'fallido', 'reembolsado') DEFAULT 'pendiente',
    
    -- Estado del pedido
    estado_pedido ENUM('pendiente', 'procesando', 'enviado', 'entregado', 'cancelado') DEFAULT 'pendiente',
    
    -- Fechas
    fecha_pedido TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    fecha_envio TIMESTAMP NULL,
    fecha_entrega TIMESTAMP NULL,
    
    notas TEXT,
    
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE RESTRICT,
    INDEX idx_usuario (id_usuario),
    INDEX idx_numero_pedido (numero_pedido),
    INDEX idx_estado (estado_pedido),
    INDEX idx_fecha (fecha_pedido)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE detalles_pedido (
    id_detalle INT AUTO_INCREMENT PRIMARY KEY,
    id_pedido INT NOT NULL,
    id_producto INT NOT NULL,
    nombre_producto VARCHAR(255) NOT NULL,
    precio_unitario DECIMAL(10, 2) NOT NULL,
    cantidad INT NOT NULL,
    subtotal DECIMAL(10, 2) NOT NULL,
    
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id_pedido) ON DELETE CASCADE,
    FOREIGN KEY (id_producto) REFERENCES productos(id_producto) ON DELETE RESTRICT,
    INDEX idx_pedido (id_pedido),
    INDEX idx_producto (id_producto)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE carrito (
    id_carrito INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT NOT NULL,
    id_producto INT NOT NULL,
    cantidad INT NOT NULL DEFAULT 1,
    fecha_agregado TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    fecha_actualizacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario) ON DELETE CASCADE,
    FOREIGN KEY (id_producto) REFERENCES productos(id_producto) ON DELETE CASCADE,
    UNIQUE KEY unique_usuario_producto (id_usuario, id_producto),
    INDEX idx_usuario (id_usuario)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE informacion_pago (
    id_pago INT AUTO_INCREMENT PRIMARY KEY,
    id_pedido INT NOT NULL,
    numero_tarjeta_enmascarado VARCHAR(20), -- Solo últimos 4 dígitos
    nombre_tarjeta VARCHAR(150),
    fecha_expiracion VARCHAR(7), -- MM/YYYY
    tipo_tarjeta VARCHAR(20), -- visa, mastercard, etc.
    fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (id_pedido) REFERENCES pedidos(id_pedido) ON DELETE CASCADE,
    INDEX idx_pedido (id_pedido)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- VISTAS ÚTILES PARA CONSULTAS

-- Vista de productos con información de categoría
CREATE VIEW vista_productos_completa AS
SELECT 
    p.id_producto,
    p.nombre,
    p.descripcion,
    p.precio,
    p.imagen,
    p.stock,
    p.destacado,
    p.activo,
    c.id_categoria,
    c.nombre_categoria,
    c.descripcion AS descripcion_categoria
FROM productos p
INNER JOIN categorias c ON p.id_categoria = c.id_categoria;

-- Vista de pedidos con información del usuario
CREATE VIEW vista_pedidos_completa AS
SELECT 
    pe.id_pedido,
    pe.numero_pedido,
    pe.nombre_cliente,
    pe.apellido_cliente,
    pe.email_cliente,
    pe.telefono_cliente,
    pe.direccion_envio,
    pe.ciudad,
    pe.codigo_postal,
    pe.pais,
    pe.subtotal,
    pe.costo_envio,
    pe.total,
    pe.metodo_pago,
    pe.estado_pago,
    pe.estado_pedido,
    pe.fecha_pedido,
    pe.fecha_envio,
    pe.fecha_entrega,
    u.username,
    u.email AS email_usuario
FROM pedidos pe
INNER JOIN usuarios u ON pe.id_usuario = u.id_usuario;

-- Vista de carrito con detalles de productos
CREATE VIEW vista_carrito_completa AS
SELECT 
    c.id_carrito,
    c.id_usuario,
    u.username,
    c.id_producto,
    p.nombre AS nombre_producto,
    p.precio,
    p.imagen,
    c.cantidad,
    (p.precio * c.cantidad) AS subtotal,
    c.fecha_agregado
FROM carrito c
INNER JOIN usuarios u ON c.id_usuario = u.id_usuario
INNER JOIN productos p ON c.id_producto = p.id_producto;

-- PROCEDIMIENTOS ALMACENADOS ÚTILES
-- Procedimiento para crear un nuevo pedido
DELIMITER //
CREATE PROCEDURE crear_pedido(
    IN p_id_usuario INT,
    IN p_nombre VARCHAR(100),
    IN p_apellido VARCHAR(100),
    IN p_email VARCHAR(100),
    IN p_telefono VARCHAR(20),
    IN p_direccion VARCHAR(255),
    IN p_ciudad VARCHAR(100),
    IN p_codigo_postal VARCHAR(10),
    IN p_pais VARCHAR(50),
    IN p_subtotal DECIMAL(10,2),
    IN p_costo_envio DECIMAL(10,2),
    IN p_total DECIMAL(10,2),
    IN p_metodo_pago VARCHAR(50),
    OUT p_id_pedido INT,
    OUT p_numero_pedido VARCHAR(20)
)
BEGIN
    -- Generar número de pedido único
    SET p_numero_pedido = CONCAT('PED', LPAD(FLOOR(RAND() * 999999), 6, '0'));
    
    -- Insertar el pedido
    INSERT INTO pedidos (
        numero_pedido, id_usuario, nombre_cliente, apellido_cliente,
        email_cliente, telefono_cliente, direccion_envio, ciudad,
        codigo_postal, pais, subtotal, costo_envio, total, metodo_pago
    ) VALUES (
        p_numero_pedido, p_id_usuario, p_nombre, p_apellido,
        p_email, p_telefono, p_direccion, p_ciudad,
        p_codigo_postal, p_pais, p_subtotal, p_costo_envio, p_total, p_metodo_pago
    );
    
    -- Obtener el ID del pedido creado
    SET p_id_pedido = LAST_INSERT_ID();
END //
DELIMITER ;

-- Procedimiento para agregar productos al carrito
DELIMITER //
CREATE PROCEDURE agregar_al_carrito(
    IN p_id_usuario INT,
    IN p_id_producto INT,
    IN p_cantidad INT
)
BEGIN
    -- Verificar si el producto ya existe en el carrito
    IF EXISTS (SELECT 1 FROM carrito WHERE id_usuario = p_id_usuario AND id_producto = p_id_producto) THEN
        -- Si existe, actualizar la cantidad
        UPDATE carrito 
        SET cantidad = cantidad + p_cantidad,
            fecha_actualizacion = CURRENT_TIMESTAMP
        WHERE id_usuario = p_id_usuario AND id_producto = p_id_producto;
    ELSE
        -- Si no existe, insertar nuevo registro
        INSERT INTO carrito (id_usuario, id_producto, cantidad)
        VALUES (p_id_usuario, p_id_producto, p_cantidad);
    END IF;
END //
DELIMITER ;
